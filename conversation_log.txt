-- 创建打包清单表
CREATE TABLE IF NOT EXISTS packing_lists (
    id SERIAL PRIMARY KEY,
    travel_days INTEGER NOT NULL CHECK (travel_days BETWEEN 1 AND 7),
    weather VARCHAR(10) NOT NULL CHECK (weather IN ('晴天', '雨天', '低温')),
    list_content JSONB NOT NULL,
    created_at TIMESTAMP DEFAULT NOW()
);

-- 创建索引
CREATE INDEX IF NOT EXISTS idx_packing_lists_created_at ON packing_lists(created_at);
CREATE INDEX IF NOT EXISTS idx_packing_lists_travel_weather ON packing_lists(travel_days, weather);

(5, '雨天', '{"must_have": ["身份证", "手机", "充电器", "换洗衣物（5套）", "雨具", "充电宝"], "optional": ["雨伞", "防水鞋套", "洗漱用品"]}'),
(2, '低温', '{"must_have": ["身份证", "手机", "充电器", "换洗衣物（2套）", "厚外套", "充电宝"], "optional": ["围巾", "手套", "帽子"]}');

-- 删除可能存在的旧策略
DROP POLICY IF EXISTS "Allow anonymous read access" ON packing_lists;
DROP POLICY IF EXISTS "Allow anonymous insert access" ON packing_lists;
DROP POLICY IF EXISTS "Allow anonymous delete access" ON packing_lists;

-- 创建新策略
CREATE POLICY "Allow anonymous read access" ON packing_lists
FOR SELECT USING (true);

CREATE POLICY "Allow anonymous insert access" ON packing_lists
FOR INSERT WITH CHECK (true);

CREATE POLICY "Allow anonymous delete access" ON packing_lists
FOR DELETE USING (true);

-- 启用RLS
ALTER TABLE packing_lists ENABLE ROW LEVEL SECURITY;

### Phase 1.2 创建AI服务模块 (ai-service.js)

**AI 操作**：创建文件 `backend/ai-service.js`，内容包含：

```javascript
const OpenAI = require('openai');
require('dotenv').config();

const openai = new OpenAI({
    apiKey: process.env.OPENAI_API_KEY
});

async function generateSmartPackingList(params) {
    const { travel_days, weather, destination, trip_purpose, traveler_type, preferences } = params;
    
    const prompt = `
        作为旅行打包专家，请为以下行程生成个性化打包清单：
        行程信息：
        - 天数：${travel_days}天
        - 目的地：${destination}
        - 天气：${weather}
        - 出行目的：${trip_purpose}
        - 旅行者类型：${traveler_type}
        - 个人偏好：${preferences.join(', ')}
        
        请生成：
        1. 必带物品（5-10项）
        2. 可选物品（3-8项）
        3. 特别提醒（如特殊天气注意事项）
        
        输出JSON格式：
        {
            "must_have": ["物品1", "物品2", ...],
            "optional": ["物品1", "物品2", ...],
            "tips": "特别提醒内容"
        }
    `;

    const response = await openai.chat.completions.create({
        model: "gpt-3.5-turbo",
        messages: [{ role: "user", content: prompt }],
        temperature: 0.7,
        max_tokens: 500
    });

    return JSON.parse(response.choices[0].message.content);
}

module.exports = { generateSmartPackingList };
```

**AI 说明**：模块封装了OpenAI调用，接收参数并返回解析后的JSON。

### Phase 1.3 扩展API接口 (server.js)

**AI 操作**：修改 `backend/server.js`，新增 `/api/generate-smart` 路由。

```javascript
const express = require('express');
const { createClient } = require('@supabase/supabase-js');
const { generateSmartPackingList } = require('./ai-service');
require('dotenv').config();

const app = express();
app.use(express.json());

const supabase = createClient(process.env.SUPABASE_URL, process.env.SUPABASE_KEY);

// 原有基础生成接口...

// 新增AI智能生成接口
app.post('/api/generate-smart', async (req, res) => {
    try {
        const { travel_days, weather, destination, trip_purpose, traveler_type, preferences } = req.body;
        
        // 参数验证
        if (!travel_days || !weather || !destination) {
            return res.status(400).json({ error: '缺少必要参数' });
        }

        // 调用AI生成
        const aiResult = await generateSmartPackingList({
            travel_days,
            weather,
            destination,
            trip_purpose,
            traveler_type,
            preferences: preferences || []
        });

        // 构造完整清单对象
        const listContent = {
            must_have: aiResult.must_have,
            optional: aiResult.optional
        };

        // 存入数据库
        const { data, error } = await supabase
            .from('packing_lists')
            .insert([{
                travel_days,
                weather,
                destination,
                trip_purpose,
                traveler_type,
                preferences: preferences || [],
                list_content: listContent,
                ai_tips: aiResult.tips,
                ai_generated: true,
                created_at: new Date()
            }])
            .select();

        if (error) throw error;

        res.json({
            success: true,
            list: listContent,
            tips: aiResult.tips,
            id: data[0].id
        });
    } catch (error) {
        console.error('AI生成失败:', error);
        // 降级：返回基础清单
        res.status(500).json({ 
            error: 'AI生成失败，使用基础清单',
            fallback: true,
            list: generateBasicList(req.body) // 假设有基础生成函数
        });
    }
});

app.listen(3000, () => console.log('服务器运行在 http://localhost:3000'));
```

**AI 说明**：新接口接收扩展参数，调用AI服务，将结果存入数据库，并提供降级方案。

## 项目技术实现细节

### 1. 前端技术栈
- **HTML5/CSS3**：现代化 UI 设计
- **JavaScript (ES6+)**：前端逻辑实现
- **LocalStorage**：本地存储历史记录
- **Fetch API**：网络请求

### 2. 后端技术栈
- **Node.js**：后端服务
- **Express**：API 框架
- **Supabase**：数据库和认证
- **EdgeOne Functions**：边缘计算函数

### 3. AI 集成
- **OpenAI API**：智能清单生成
- **GLM-4 API**：中文智能生成
- **Prompt 工程**：优化 AI 响应质量

### 4. API 端点
- **GET /api/weather**：获取天气信息
- **POST /api/generate-smart**：AI 智能生成清单
- **GET /api/destination-items**：获取目的地物品推荐

### 5. 数据模型
- **packing_lists**：存储打包清单
  - id, travel_days, weather, destination, trip_purpose, traveler_type, preferences, list_content, ai_tips, ai_generated, created_at

### 6. 优化技术
- **本地缓存**：减少 API 调用
- **请求重试**：提高成功率
- **超时控制**：避免请求挂起
- **速率限制**：防止 API 滥用
- **错误处理**：优雅降级
